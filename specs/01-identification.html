1 Principes generaux

Trois entites ont un identifiant metier visible : Projet, Contrat et Facture. Tout le reste est identifie en interne par UUID.

RegleDetail

Identifiant metierColonne ref — affiche dans l'interface, les URL, les exports. Genere automatiquement a la creation.
ImmutabiliteUne ref ne change jamais, meme si l'entite est modifiee ou archivee.
UniciteContrainte UNIQUE sur la colonne ref.

4 Identifiant Facture

Numerotation sequentielle globale, sans trou (obligation legale francaise).

FAC
Prefixe
-
DUP
Trigramme client
-
0012
Numero global (4 digits)
ProprieteDetail
CompteurSequentiel global, tous clients confondus. De 0001 a 9999.
TrigrammeReprend le trigramme du client associe a la facture (meme que celui du projet).
Sans trouObligation legale. En cas d'annulation, on cree un avoir — le numero de la facture n'est jamais reutilise.
OdooLa ref FAC-DUP-0012 est transmise a Odoo comme reference externe dans account.move.
GenerationA la creation de la facture par un Admin dans la page Comptabilite.

⚠

Obligation legale — La numerotation est strictement croissante et sans trou. Le compteur global (0001, 0002, 0003...) garantit cette continuite quel que soit le client. Supprimer une facture est interdit : on emet un avoir.

7 Generation technique

Implementation PostgreSQL via Supabase : sequences, tables et triggers.

Table des typologies

-- Table dynamique, gerable par l'admin
CREATE TABLE typologies_projet (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  code TEXT UNIQUE NOT NULL, -- ex: 'APP', 'POE', 'PDC'
  libelle TEXT NOT NULL, -- ex: 'Apprentissage'
  created_at TIMESTAMPTZ DEFAULT NOW()
);

Table des clients (trigramme)

-- Le trigramme est auto-genere puis modifiable par l'admin
ALTER TABLE clients
  ADD COLUMN trigramme TEXT UNIQUE NOT NULL;
-- Index pour verifier les collisions rapidement

Sequences

CREATE SEQUENCE seq_projet_ref START 1 INCREMENT 1;
CREATE SEQUENCE seq_contrat_ref START 1 INCREMENT 1;
CREATE SEQUENCE seq_facture_ref START 1 INCREMENT 1;

Trigger — Projet

CREATE OR REPLACE FUNCTION generate_projet_ref()
RETURNS TRIGGER AS $$
DECLARE
  v_trigramme TEXT;
  v_typo TEXT;
BEGIN
  SELECT trigramme INTO v_trigramme FROM clients WHERE id = NEW.client_id;
  SELECT code INTO v_typo FROM typologies_projet WHERE id = NEW.typologie_id;
  NEW.ref := LPAD(nextval('seq_projet_ref')::TEXT, 4, '0')
    || '-' || v_trigramme
    || '-' || v_typo;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_projet_ref BEFORE INSERT ON projets
  FOR EACH ROW EXECUTE FUNCTION generate_projet_ref();

Trigger — Contrat

CREATE OR REPLACE FUNCTION generate_contrat_ref()
RETURNS TRIGGER AS $$
BEGIN
  NEW.ref := 'CTR-' || LPAD(nextval('seq_contrat_ref')::TEXT, 5, '0');
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_contrat_ref BEFORE INSERT ON contrats
  FOR EACH ROW EXECUTE FUNCTION generate_contrat_ref();

Trigger — Facture

CREATE OR REPLACE FUNCTION generate_facture_ref()
RETURNS TRIGGER AS $$
DECLARE
  v_trigramme TEXT;
BEGIN
  SELECT c.trigramme INTO v_trigramme
  FROM clients c WHERE c.id = NEW.client_id;

  NEW.ref := 'FAC-' || v_trigramme || '-'
    || LPAD(nextval('seq_facture_ref')::TEXT, 4, '0');
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_facture_ref BEFORE INSERT ON factures
  FOR EACH ROW EXECUTE FUNCTION generate_facture_ref();

Trigger — Trigramme client (auto-generation)

CREATE OR REPLACE FUNCTION generate_trigramme()
RETURNS TRIGGER AS $$
DECLARE
  v_tri TEXT;
BEGIN
  -- Prend les 3 premieres lettres en majuscules
  v_tri := UPPER(LEFT(NEW.nom, 3));

  -- Si collision, prendre 4 lettres
  IF EXISTS (SELECT 1 FROM clients WHERE trigramme = v_tri) THEN
    v_tri := UPPER(LEFT(NEW.nom, 4));
  END IF;

  NEW.trigramme := v_tri;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_client_trigramme BEFORE INSERT ON clients
  FOR EACH ROW EXECUTE FUNCTION generate_trigramme();
